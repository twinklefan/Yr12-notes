Databases

    Examples - Basics

  • SELECT surname, first_name
    FROM people
  • SELECT first_name, surname
    FROM people
    WHERE DOB = 1991;
  • SELECT first_name
    FROM people
    WHERE first_name > 'a%'; (starts with a)
  • SELECT *
    FROM people
    WHERE origin = 'France'
    OR origin = 'Germany';
  • SELECT first_name, last_name
    FROM people
    WHERE age > 21
    AND (origin = 'France'
    OR origin = 'Germany');
  • SELECT first_name, origin
    FROM people
    WHERE DOB < 2000
    AND origin = 'Australia'
    ORDER BY last_name DESC; (orders it in descending order)
  • SELECT first_name, last_name, DOB
    FROM people
    ORDER BY DOB;
  • SELECT country, gold, bronze, silver
    FROM countryMedal
    WHERE gold > 0
    ORDER BY gold DESC, bronze DESC, silver DESC, country DESC;

    • Examples - Joins

  • SELECT title, name
    FROM author
    JOIN book
    ON author.id = book.author_id
  • SELECT title, rating
    FROM author
    INNER JOIN book
    ON author.id = book.author_id
    WHERE author.death_year IS NULL
  • SELECT author.name, book.title, book.publish_year
    FROM author
    JOIN book
    ON author.id = book.author_id
    WHERE book.publish_year > 2005;
  • SELECT book.title AS book_title, adaptation.title AS adaptation_title, adaptation.release_year,book.publish_year
    FROM book
    JOIN adaptation
    ON book.id = adaptation.book_id
    WHERE book.rating < adaptation.rating
    AND (adaptation.release_year - book.publish_year) < 5;
